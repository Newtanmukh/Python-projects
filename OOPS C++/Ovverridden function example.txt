// C++ program to access overridden function
// in main() using the scope resolution operator ::

#include <iostream>
using namespace std;

class A {
   public:
    void print() {
        cout << "Class A" << endl;
    }
};

class B {
   public:
    void print() {
        cout << "Class B" << endl;
    }
};

class C:public A,public B
{    public:
    void test()
    {
        B::print();
    }
};

int main() {
    C c;
    c.test();
    c.A::print();
    return 0;
}
